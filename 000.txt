CREATE TABLE users (
    id SERIAL PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    surname VARCHAR(100) NOT NULL,
    phone VARCHAR(20) NOT NULL,
    email VARCHAR(100) NOT NULL,
    login VARCHAR(50) UNIQUE NOT NULL,
    password VARCHAR(100) NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
); 

CREATE TABLE orders (
    id SERIAL PRIMARY KEY,
    user_id INTEGER NOT NULL REFERENCES users(id) ON DELETE CASCADE,
    reservation_datetime TIMESTAMP NOT NULL,
    guests INTEGER NOT NULL CHECK (guests BETWEEN 1 AND 10),
    phone VARCHAR(20) NOT NULL,
    status VARCHAR(50) DEFAULT 'новая' CHECK (status IN ('новая', 'посещение состоялось', 'отменена')),
    review_rating INTEGER CHECK (review_rating BETWEEN 1 AND 5),
    review_text TEXT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
